89  template<typename T> void npy_save(std::string fname, const T* data, const std::vector<size_t> shape, std::string mode = "w") {
90  FILE* fp = NULL;
91  std::vector<size_t> true_data_shape; //if appending, the shape of existing + new data
92 
93  if(mode == "a") fp = fopen(fname.c_str(),"r+b");
94 
95  if(fp) {
96  //file exists. we need to append to it. read the header, modify the array size
97  size_t word_size;
98  bool fortran_order;
99  parse_npy_header(fp,word_size,true_data_shape,fortran_order);
100  assert(!fortran_order);
101 
102  if(word_size != sizeof(T)) {
103  std::cout<<"libnpy error: "<<fname<<" has word size "<<word_size<<" but npy_save appending data sized "<<sizeof(T)<<"\n";
104  assert( word_size == sizeof(T) );
105  }
106  if(true_data_shape.size() != shape.size()) {
107  std::cout<<"libnpy error: npy_save attempting to append misdimensioned data to "<<fname<<"\n";
108  assert(true_data_shape.size() != shape.size());
109  }
110 
111  for(size_t i = 1; i < shape.size(); i++) {
112  if(shape[i] != true_data_shape[i]) {
113  std::cout<<"libnpy error: npy_save attempting to append misshaped data to "<<fname<<"\n";
114  assert(shape[i] == true_data_shape[i]);
115  }
116  }
117  true_data_shape[0] += shape[0];
118  }
119  else {
120  fp = fopen(fname.c_str(),"wb");
121  true_data_shape = shape;
122  }
123 
124  std::vector<char> header = create_npy_header<T>(true_data_shape);
125  size_t nels = std::accumulate(shape.begin(),shape.end(),1,std::multiplies<size_t>());
126 
127  fseek(fp,0,SEEK_SET);
128  fwrite(&header[0],sizeof(char),header.size(),fp);
129  fseek(fp,0,SEEK_END);
130  fwrite(data,sizeof(T),nels,fp);
131  fclose(fp);
132  }

134  template<typename T> void npz_save(std::string zipname, std::string fname, const T* data, const std::vector<size_t>& shape, std::string mode = "w")
135  {
136  //first, append a .npy to the fname
137  fname += ".npy";
138 
139  //now, on with the show
140  FILE* fp = NULL;
141  uint16_t nrecs = 0;
142  size_t global_header_offset = 0;
143  std::vector<char> global_header;
144 
145  if(mode == "a") fp = fopen(zipname.c_str(),"r+b");
146 
147  if(fp) {
148  //zip file exists. we need to add a new npy file to it.
149  //first read the footer. this gives us the offset and size of the global header
150  //then read and store the global header.
151  //below, we will write the the new data at the start of the global header then append the global header and footer below it
152  size_t global_header_size;
153  parse_zip_footer(fp,nrecs,global_header_size,global_header_offset);
154  fseek(fp,global_header_offset,SEEK_SET);
155  global_header.resize(global_header_size);
156  size_t res = fread(&global_header[0],sizeof(char),global_header_size,fp);
157  if(res != global_header_size){
158  throw std::runtime_error("npz_save: header read error while adding to existing zip");
159  }
160  fseek(fp,global_header_offset,SEEK_SET);
161  }
162  else {
163  fp = fopen(zipname.c_str(),"wb");
164  }
165 
166  std::vector<char> npy_header = create_npy_header<T>(shape);
167 
168  size_t nels = std::accumulate(shape.begin(),shape.end(),1,std::multiplies<size_t>());
169  size_t nbytes = nels*sizeof(T) + npy_header.size();
170 
171  //get the CRC of the data to be added
172  uint32_t crc = crc32(0L,(uint8_t*)&npy_header[0],npy_header.size());
173  crc = crc32(crc,(uint8_t*)data,nels*sizeof(T));
174 
175  //build the local header
176  std::vector<char> local_header;
177  local_header += "PK"; //first part of sig
178  local_header += (uint16_t) 0x0403; //second part of sig
179  local_header += (uint16_t) 20; //min version to extract
180  local_header += (uint16_t) 0; //general purpose bit flag
181  local_header += (uint16_t) 0; //compression method
182  local_header += (uint16_t) 0; //file last mod time
183  local_header += (uint16_t) 0; //file last mod date
184  local_header += (uint32_t) crc; //crc
185  local_header += (uint32_t) nbytes; //compressed size
186  local_header += (uint32_t) nbytes; //uncompressed size
187  local_header += (uint16_t) fname.size(); //fname length
188  local_header += (uint16_t) 0; //extra field length
189  local_header += fname;
190 
191  //build global header
192  global_header += "PK"; //first part of sig
193  global_header += (uint16_t) 0x0201; //second part of sig
194  global_header += (uint16_t) 20; //version made by
195  global_header.insert(global_header.end(),local_header.begin()+4,local_header.begin()+30);
196  global_header += (uint16_t) 0; //file comment length
197  global_header += (uint16_t) 0; //disk number where file starts
198  global_header += (uint16_t) 0; //internal file attributes
199  global_header += (uint32_t) 0; //external file attributes
200  global_header += (uint32_t) global_header_offset; //relative offset of local file header, since it begins where the global header used to begin
201  global_header += fname;
202 
203  //build footer
204  std::vector<char> footer;
205  footer += "PK"; //first part of sig
206  footer += (uint16_t) 0x0605; //second part of sig
207  footer += (uint16_t) 0; //number of this disk
208  footer += (uint16_t) 0; //disk where footer starts
209  footer += (uint16_t) (nrecs+1); //number of records on this disk
210  footer += (uint16_t) (nrecs+1); //total number of records
211  footer += (uint32_t) global_header.size(); //nbytes of global headers
212  footer += (uint32_t) (global_header_offset + nbytes + local_header.size()); //offset of start of global headers, since global header now starts after newly written array
213  footer += (uint16_t) 0; //zip file comment length
214 
215  //write everything
216  fwrite(&local_header[0],sizeof(char),local_header.size(),fp);
217  fwrite(&npy_header[0],sizeof(char),npy_header.size(),fp);
218  fwrite(data,sizeof(T),nels,fp);
219  fwrite(&global_header[0],sizeof(char),global_header.size(),fp);
220  fwrite(&footer[0],sizeof(char),footer.size(),fp);
221  fclose(fp);
222  }

236  template<typename T> std::vector<char> create_npy_header(const std::vector<size_t>& shape) {
237 
238  std::vector<char> dict;
239  dict += "{'descr': '";
240  dict += BigEndianTest();
241  dict += map_type(typeid(T));
242  dict += std::to_string(sizeof(T));
243  dict += "', 'fortran_order': False, 'shape': (";
244  dict += std::to_string(shape[0]);
245  for(size_t i = 1;i < shape.size();i++) {
246  dict += ", ";
247  dict += std::to_string(shape[i]);
248  }
249  if(shape.size() == 1) dict += ",";
250  dict += "), }";
251  //pad with spaces so that preamble+dict is modulo 16 bytes. preamble is 10 bytes. dict needs to end with \n
252  int remainder = 16 - (10 + dict.size()) % 16;
253  dict.insert(dict.end(),remainder,' ');
254  dict.back() = '\n';
255 
256  std::vector<char> header;
257  header += (char) 0x93;
258  header += "NUMPY";
259  header += (char) 0x01; //major version of numpy format
260  header += (char) 0x00; //minor version of numpy format
261  header += (uint16_t) dict.size();
262  header.insert(header.end(),dict.begin(),dict.end());
263 
264  return header;
265  }
